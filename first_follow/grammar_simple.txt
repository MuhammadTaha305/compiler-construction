# Simple arithmetic grammar (no left recursion)
# S -> E
# E -> E + T | E - T | T
# (Rewritten to eliminate left recursion)
# E -> T E'
# E' -> + T E' | - T E' | epsilon
# T -> T * F | T / F | F
# (Rewritten)
# T -> F T'
# T' -> * F T' | / F T' | epsilon
# F -> ( E ) | id | num
